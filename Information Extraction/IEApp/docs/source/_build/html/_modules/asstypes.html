[docs]	def foodie(self, data, assertions):
		"""
		Identifying assertion type for The Foodie

		:param assertions: array of extracted information
		:param data: array of data frames
	    :return: array of The Foodie dataframes
	    """

		for d in data:
			if not d['Description'] == "" and d['Description'] is not None and (d['Food'] is not None or len(d['Organization']) != 0):
				assertions.append({
					'type' : "food_describe",
					'values' : {
						"Person" 			: d['Person'],
						"Food" 				: d['Food'],
						"Description" 		: d['Description'],
						"Organization" 		: d['Organization'],
						"Type" 				: d['Type'],
						"Location" 			: d['Location']
					},
					'persona' : "The Foodie",
					'date' : d['Date'],
					'time' : d['Time']
				})
			if not d['Action'] == "" and d['Action'] is not None and (d['Food'] is not None or len(d['Organization']) != 0):
				assertions.append({
					'type' : "food_activity",
					'values' : {
						"Person" 			: d['Person'],
						"Food" 				: d['Food'],
						"Action" 			: d['Action'],
						"Organization" 		: d['Organization'],
						"Type" 				: d['Type'],

						"Location" 			: d['Location'],
						'date' : d['Date'],
						'time' : d['Time']
					},
					'persona' : "The Foodie"

			if not d['Sentiment'] == "" and d['Sentiment'] is not None and (d['Food'] is not None or len(d['Organization']) != 0):

					},
					'persona' : "The Foodie"
				})
		return assertions
