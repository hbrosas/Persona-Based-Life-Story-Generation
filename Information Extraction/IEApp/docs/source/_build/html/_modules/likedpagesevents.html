[docs]	def getPageName(self, page):
		"""
		gets the page name from a liked page

		:param page: page
	    :return: page name
	    """

		doc = self.nlp(page)
		page = ""

		for sent in doc.sents:
			result = re.search("He likes (.*)", sent.text)
			if result is not None:
				page = result.group(1)
			break

		return page

		return assertions



[docs]	def getEventDetails(self, page):
		"""
		gets the event details such as event name and rsvp from the event

		:param page: page
	    :return: event, rsvp
	    """

		doc = self.nlp(page)
		str = page.split('. ')
		event = ""
		rsvp = ""

		result = re.search("He is (.*) in (.*)", str[0])
		if result is not None:
			rsvp = result.group(1)
			event = result.group(2)

		return event, rsvp
